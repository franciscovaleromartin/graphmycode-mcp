```typescript
// src/lib/codeflow/astToMermaid.ts
import type { ParsedFlow, FlowNode, FlowNodeType } from './types';

const safeId = (id: string) => id.replace(/[^a-zA-Z0-9_]/g, '_');
const safeLabel = (label: string) => label.replace(/["\[\]<>{}()]/g, (c) => {
  const map: Record<string, string> = { '"': "'", '[': '(', ']': ')', '<': '', '>': '', '{': '', '}': '', '(': '', ')': '' };
  return map[c] ?? c;
}).substring(0, 40);

function nodeShape(node: FlowNode): string {
  const l = safeLabel(node.label);
  switch (node.type) {
    case 'condition': return `{"${l}"}`;
    case 'loop':      return `{{"${l}"}}`;
    case 'error':     return `[/"${l}"/]`;
    default:          return `["${l}"]`;
  }
}

const CLASS_MAP: Record<FlowNodeType, string> = {
  function: 'fn',
  condition: 'cond',
  loop: 'loop',
  error: 'err',
  class: 'cls',
  element: 'el',
  special: 'special',
  selector: 'sel',
  media: 'media',
  key: 'key',
};

export function astToMermaid(flow: ParsedFlow): string {
  const { nodes, edges, language } = flow;

  if (nodes.length === 0) {
    return 'graph TD\n  empty["No structures detected"]';
  }

  const lines: string[] = [];

  if (language === 'css') {
    lines.push('graph LR');
  } else if (language === 'js-ts' || language === 'python') {
    lines.push('flowchart TD');
  } else {
    lines.push('graph TD');
  }

  lines.push('  classDef fn fill:#1e293b,stroke:#22d3ee,stroke-width:2px,color:#f1f5f9;');
  lines.push('  classDef cond fill:#1e293b,stroke:#f59e0b,stroke-width:2px,color:#f1f5f9;');
  lines.push('  classDef loop fill:#1e293b,stroke:#8b5cf6,stroke-width:2px,color:#f1f5f9;');
  lines.push('  classDef err fill:#1e293b,stroke:#ef4444,stroke-width:2px,color:#f1f5f9;');
  lines.push('  classDef cls fill:#0f172a,stroke:#475569,stroke-width:2px,color:#94a3b8;');
  lines.push('  classDef el fill:#1e293b,stroke:#34d399,stroke-width:1px,color:#f1f5f9;');
  lines.push('  classDef special fill:#1e293b,stroke:#fbbf24,stroke-width:2px,color:#f1f5f9;');
  lines.push('  classDef sel fill:#1e293b,stroke:#f472b6,stroke-width:1px,color:#f1f5f9;');
  lines.push('  classDef media fill:#0f172a,stroke:#f472b6,stroke-width:2px,color:#94a3b8;');
  lines.push('  classDef key fill:#1e293b,stroke:#60a5fa,stroke-width:1px,color:#f1f5f9;');

  const classNodes = nodes.filter((n) => n.type === 'class');
  const childIds = new Set(nodes.filter((n) => n.parentId).map((n) => n.id));
  const standaloneNodes = nodes.filter((n) => !n.parentId && n.type !== 'class');

  for (const node of standaloneNodes) {
    lines.push(`  ${safeId(node.id)}${nodeShape(node)}:::${CLASS_MAP[node.type] ?? 'fn'}`);
  }

  for (const cls of classNodes) {
    lines.push(`  subgraph ${safeId(cls.id)}["${safeLabel(cls.label)}"]`);
    const children = nodes.filter((n) => n.parentId === cls.id);
    for (const child of children) {
      lines.push(`    ${safeId(child.id)}${nodeShape(child)}:::${CLASS_MAP[child.type] ?? 'fn'}`);
    }
    lines.push('  end');
  }

  for (const edge of edges) {
    lines.push(`  ${safeId(edge.from)} --> ${safeId(edge.to)}`);
  }

  return lines.join('\n');
}
```

```typescript
// src/lib/codeflow/parsers/html-parser.ts
import type { ParsedFlow, FlowNode, FlowEdge } from '../types';

const VOID_ELEMENTS = new Set([
  'area','base','br','col','embed','hr','img','input',
  'link','meta','param','source','track','wbr',
]);
const SPECIAL_ELEMENTS = new Set(['script', 'link', 'style']);

export function parseHtml(content: string): ParsedFlow {
  const nodes: FlowNode[] = [];
  const edges: FlowEdge[] = [];
  let nodeCount = 0;
  const stack: string[] = [];

  const tokenRegex = /<(\/?)([a-zA-Z][a-zA-Z0-9-]*)([^>]*?)(\/?)>/g;
  let match: RegExpExecArray | null;

  while ((match = tokenRegex.exec(content)) !== null) {
    const [, closing, rawTag, attrs, selfClosing] = match;
    const tag = rawTag.toLowerCase();

    if (closing) {
      if (!VOID_ELEMENTS.has(tag)) stack.pop();
      continue;
    }

    const id = `el_${nodeCount++}`;
    const idAttr = attrs.match(/\bid=["']([^"']+)["']/)?.[1];
    const classAttr = attrs.match(/\bclass=["']([^"']*?)["']/)?.[1]?.trim().split(/\s+/)[0];
    const label = idAttr ? `<${tag}#${idAttr}>`
                : classAttr ? `<${tag}.${classAttr}>`
                : `<${tag}>`;

    const type = SPECIAL_ELEMENTS.has(tag) ? 'special' : 'element';
    nodes.push({ id, label, type });

    if (stack.length > 0) {
      edges.push({ from: stack[stack.length - 1], to: id });
    }

    if (!selfClosing && !VOID_ELEMENTS.has(tag)) {
      stack.push(id);
    }
  }

  return { nodes, edges, language: 'html', truncated: false };
}
```

```typescript
// src/lib/codeflow/parsers/css-parser.ts
import type { ParsedFlow, FlowNode, FlowEdge } from '../types';

export function parseCss(content: string): ParsedFlow {
  const nodes: FlowNode[] = [];
  const edges: FlowEdge[] = [];
  let nodeCount = 0;

  // Extract @media blocks first
  const mediaRegex = /@media\s*([^{]+)\{([\s\S]*?)(?=@media|\s*$)/g;
  let mediaMatch: RegExpExecArray | null;
  const mediaRanges: [number, number][] = [];

  while ((mediaMatch = mediaRegex.exec(content)) !== null) {
    const mediaId = `media_${nodeCount++}`;
    const mediaLabel = `@media ${mediaMatch[1].trim().substring(0, 30)}`;
    nodes.push({ id: mediaId, label: mediaLabel, type: 'media' });
    mediaRanges.push([mediaMatch.index, mediaMatch.index + mediaMatch[0].length]);

    // Extract selectors inside this media block
    const innerContent = mediaMatch[2];
    const innerSels = extractSelectors(innerContent, nodeCount);
    nodeCount += innerSels.nodes.length;
    for (const n of innerSels.nodes) {
      n.parentId = mediaId;
      nodes.push(n);
    }
    edges.push(...innerSels.edges);
  }

  // Extract :root variables as a special node
  const rootVarRegex = /:root\s*\{([^}]+)\}/g;
  let rootMatch: RegExpExecArray | null;
  while ((rootMatch = rootVarRegex.exec(content)) !== null) {
    const vars = rootMatch[1].match(/--[a-zA-Z-]+/g) ?? [];
    if (vars.length > 0) {
      const id = `special_${nodeCount++}`;
      nodes.push({ id, label: `:root (${vars.length} vars)`, type: 'special' });
    }
  }

  // Extract top-level selectors (outside media blocks)
  let strippedContent = content;
  for (const [start, end] of mediaRanges.reverse()) {
    strippedContent = strippedContent.slice(0, start) + strippedContent.slice(end);
  }
  // Remove :root block
  strippedContent = strippedContent.replace(/:root\s*\{[^}]+\}/g, '');

  const topLevel = extractSelectors(strippedContent, nodeCount);
  nodes.push(...topLevel.nodes);
  edges.push(...topLevel.edges);

  return { nodes, edges, language: 'css', truncated: false };
}

function extractSelectors(content: string, startCount: number): { nodes: FlowNode[]; edges: FlowEdge[] } {
  const nodes: FlowNode[] = [];
  const edges: FlowEdge[] = [];
  let nodeCount = startCount;

  const selectorRegex = /([.#]?[a-zA-Z][a-zA-Z0-9_-]*(?:[:.][a-zA-Z][a-zA-Z0-9_-]*)*)\s*\{[^}]*\}/g;
  let match: RegExpExecArray | null;

  while ((match = selectorRegex.exec(content)) !== null) {
    const selector = match[1].trim();
    if (!selector || selector.startsWith('@') || selector === 'root') continue;
    const id = `sel_${nodeCount++}`;
    nodes.push({ id, label: selector, type: 'selector' });
  }

  return { nodes, edges };
}
```

```typescript
// src/lib/codeflow/parsers/json-yaml-parser.ts
import type { ParsedFlow, FlowNode, FlowEdge } from '../types';

export function parseJsonYaml(content: string, filePath: string): ParsedFlow {
  const ext = filePath.split('.').pop()?.toLowerCase();
  if (ext === 'json') return parseJson(content);
  return parseYaml(content);
}

function parseJson(content: string): ParsedFlow {
  const nodes: FlowNode[] = [];
  const edges: FlowEdge[] = [];
  let nodeCount = 0;

  let data: unknown;
  try {
    data = JSON.parse(content);
  } catch {
    return { nodes: [], edges: [], language: 'json-yaml', truncated: false };
  }

  function walk(value: unknown, label: string, parentId: string | null): void {
    const id = `key_${nodeCount++}`;
    const isComplex = typeof value === 'object' && value !== null;
    const displayLabel = isComplex
      ? label
      : `${label}: ${String(value).substring(0, 30)}`;

    nodes.push({ id, label: displayLabel, type: 'key' });
    if (parentId) edges.push({ from: parentId, to: id });

    if (isComplex) {
      if (Array.isArray(value)) {
        value.forEach((item, i) => walk(item, `[${i}]`, id));
      } else {
        Object.entries(value as Record<string, unknown>).forEach(([k, v]) => walk(v, k, id));
      }
    }
  }

  walk(data, 'root', null);
  return { nodes, edges, language: 'json-yaml', truncated: false };
}

function parseYaml(content: string): ParsedFlow {
  const nodes: FlowNode[] = [];
  const edges: FlowEdge[] = [];
  let nodeCount = 0;
  const stack: { id: string; indent: number }[] = [];

  for (const rawLine of content.split('\n')) {
    const line = rawLine.replace(/\t/g, '  ');
    if (line.trim() === '' || line.trim().startsWith('#')) continue;

    const indent = line.search(/\S/);
    if (indent < 0) continue;

    const keyMatch = line.trim().match(/^([^:]+):\s*(.*)$/);
    if (!keyMatch) continue;

    const [, key, value] = keyMatch;
    const trimmedValue = value.trim();
    const label = trimmedValue
      ? `${key.trim()}: ${trimmedValue.substring(0, 30)}`
      : key.trim();

    const id = `key_${nodeCount++}`;
    nodes.push({ id, label, type: 'key' });

    while (stack.length > 0 && stack[stack.length - 1].indent >= indent) {
      stack.pop();
    }
    if (stack.length > 0) {
      edges.push({ from: stack[stack.length - 1].id, to: id });
    }
    stack.push({ id, indent });
  }

  return { nodes, edges, language: 'json-yaml', truncated: false };
}
```

```typescript
// src/lib/codeflow/parsers/js-ts-parser.ts
import type { ParsedFlow, FlowNode, FlowEdge } from '../types';
import { loadParser, loadLanguage } from '../../../core/tree-sitter/parser-loader';
import { SupportedLanguages } from '../../../config/supported-languages';

const MAX_AST_NODES = 500;

type SyntaxNode = {
  type: string;
  text: string;
  namedChildren: SyntaxNode[];
  childForFieldName: (name: string) => SyntaxNode | null;
};

const FUNCTION_TYPES = new Set([
  'function_declaration', 'function_expression', 'arrow_function',
  'method_definition', 'function',
]);

function sanitizeId(text: string, suffix: number): string {
  return `${text.replace(/[^a-zA-Z0-9_]/g, '_')}_${suffix}`;
}

function countNodes(node: SyntaxNode): number {
  let count = 1;
  for (const child of node.namedChildren) count += countNodes(child);
  return count;
}

interface WalkCtx {
  nodes: FlowNode[];
  edges: FlowEdge[];
  fnIds: Map<string, string>;
  counter: number;
}

function collectFunctions(node: SyntaxNode, ctx: WalkCtx): void {
  if (FUNCTION_TYPES.has(node.type)) {
    const nameNode = node.childForFieldName('name') ?? node.childForFieldName('key');
    const name = nameNode?.text;
    if (name) {
      const id = sanitizeId(`fn_${name}`, ctx.counter++);
      ctx.nodes.push({ id, label: `${name}()`, type: 'function' });
      ctx.fnIds.set(name, id);
    }
  }
  for (const child of node.namedChildren) collectFunctions(child, ctx);
}

function collectCallsAndFlow(node: SyntaxNode, ctx: WalkCtx, currentFnId: string | null): void {
  const { type } = node;

  if (FUNCTION_TYPES.has(type)) {
    const nameNode = node.childForFieldName('name') ?? node.childForFieldName('key');
    const fnId = nameNode?.text ? ctx.fnIds.get(nameNode.text) ?? null : null;
    for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, fnId);
    return;
  }

  if (type === 'call_expression' && currentFnId) {
    const callee = node.childForFieldName('function');
    const calleeName =
      callee?.type === 'identifier' ? callee.text
      : callee?.type === 'member_expression' ? callee.childForFieldName('property')?.text
      : null;
    if (calleeName && ctx.fnIds.has(calleeName)) {
      const targetId = ctx.fnIds.get(calleeName)!;
      if (targetId !== currentFnId && !ctx.edges.some((e) => e.from === currentFnId && e.to === targetId)) {
        ctx.edges.push({ from: currentFnId, to: targetId });
      }
    }
  }

  if (currentFnId) {
    if (type === 'if_statement') {
      const id = `cond_${ctx.counter++}`;
      ctx.nodes.push({ id, label: 'if', type: 'condition' });
      ctx.edges.push({ from: currentFnId, to: id });
      for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
      return;
    }
    if (['for_statement', 'for_in_statement', 'for_of_statement', 'while_statement'].includes(type)) {
      const id = `loop_${ctx.counter++}`;
      ctx.nodes.push({ id, label: type.startsWith('for') ? 'for' : 'while', type: 'loop' });
      ctx.edges.push({ from: currentFnId, to: id });
      for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
      return;
    }
    if (type === 'try_statement') {
      const id = `err_${ctx.counter++}`;
      ctx.nodes.push({ id, label: 'try/catch', type: 'error' });
      ctx.edges.push({ from: currentFnId, to: id });
      for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
      return;
    }
  }

  for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
}

export function extractFlowFromAst(rootNode: SyntaxNode, truncateOnly: boolean): Omit<ParsedFlow, 'language' | 'truncated'> {
  const ctx: WalkCtx = { nodes: [], edges: [], fnIds: new Map(), counter: 0 };
  collectFunctions(rootNode, ctx);
  if (!truncateOnly) {
    collectCallsAndFlow(rootNode, ctx, null);
  }
  return { nodes: ctx.nodes, edges: ctx.edges };
}

export async function parseJsTs(content: string, filePath: string): Promise<ParsedFlow> {
  const ext = filePath.split('.').pop()?.toLowerCase() ?? '';
  const lang = ['js', 'jsx', 'mjs', 'cjs'].includes(ext)
    ? SupportedLanguages.JavaScript
    : SupportedLanguages.TypeScript;

  const parser = await loadParser();
  await loadLanguage(lang, filePath);
  const tree = parser.parse(content);

  const totalNodes = countNodes(tree.rootNode as unknown as SyntaxNode);
  const truncated = totalNodes > MAX_AST_NODES;

  const { nodes, edges } = extractFlowFromAst(tree.rootNode as unknown as SyntaxNode, truncated);
  return { nodes, edges, language: 'js-ts', truncated };
}
```

```typescript
// src/lib/codeflow/parsers/python-parser.ts
import type { ParsedFlow, FlowNode, FlowEdge } from '../types';
import { loadParser, loadLanguage } from '../../../core/tree-sitter/parser-loader';
import { SupportedLanguages } from '../../../config/supported-languages';

const MAX_AST_NODES = 500;

type SyntaxNode = {
  type: string;
  text: string;
  namedChildren: SyntaxNode[];
  childForFieldName: (name: string) => SyntaxNode | null;
};

interface WalkCtx {
  nodes: FlowNode[];
  edges: FlowEdge[];
  fnIds: Map<string, string>;
  counter: number;
}

function countNodes(node: SyntaxNode): number {
  let count = 1;
  for (const child of node.namedChildren) count += countNodes(child);
  return count;
}

function collectFunctions(node: SyntaxNode, ctx: WalkCtx, parentClassId?: string): void {
  if (node.type === 'class_definition') {
    const name = node.childForFieldName('name')?.text;
    if (name) {
      const id = `cls_${name.replace(/[^a-zA-Z0-9_]/g, '_')}_${ctx.counter++}`;
      ctx.nodes.push({ id, label: name, type: 'class' });
      const body = node.childForFieldName('body');
      if (body) {
        for (const child of body.namedChildren) collectFunctions(child, ctx, id);
      }
      return;
    }
  }
  if (node.type === 'function_definition') {
    const name = node.childForFieldName('name')?.text;
    if (name) {
      const id = `fn_${name.replace(/[^a-zA-Z0-9_]/g, '_')}_${ctx.counter++}`;
      const fnNode: FlowNode = { id, label: `${name}()`, type: 'function' };
      if (parentClassId) fnNode.parentId = parentClassId;
      ctx.nodes.push(fnNode);
      ctx.fnIds.set(name, id);
    }
    return;
  }
  for (const child of node.namedChildren) collectFunctions(child, ctx, parentClassId);
}

function collectCallsAndFlow(node: SyntaxNode, ctx: WalkCtx, currentFnId: string | null): void {
  const { type } = node;

  if (type === 'function_definition') {
    const name = node.childForFieldName('name')?.text;
    const fnId = name ? ctx.fnIds.get(name) ?? null : null;
    const body = node.childForFieldName('body');
    if (body) {
      for (const child of body.namedChildren) collectCallsAndFlow(child, ctx, fnId);
    }
    return;
  }

  if (type === 'call' && currentFnId) {
    const fnField = node.childForFieldName('function');
    const calleeName = fnField?.type === 'identifier' ? fnField.text
      : fnField?.type === 'attribute' ? fnField.childForFieldName('attribute')?.text
      : null;
    if (calleeName && ctx.fnIds.has(calleeName)) {
      const targetId = ctx.fnIds.get(calleeName)!;
      if (targetId !== currentFnId && !ctx.edges.some((e) => e.from === currentFnId && e.to === targetId)) {
        ctx.edges.push({ from: currentFnId, to: targetId });
      }
    }
  }

  if (currentFnId) {
    if (type === 'if_statement') {
      const id = `cond_${ctx.counter++}`;
      ctx.nodes.push({ id, label: 'if', type: 'condition' });
      ctx.edges.push({ from: currentFnId, to: id });
      for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
      return;
    }
    if (type === 'for_statement' || type === 'while_statement') {
      const id = `loop_${ctx.counter++}`;
      ctx.nodes.push({ id, label: type === 'for_statement' ? 'for' : 'while', type: 'loop' });
      ctx.edges.push({ from: currentFnId, to: id });
      for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
      return;
    }
    if (type === 'try_statement') {
      const id = `err_${ctx.counter++}`;
      ctx.nodes.push({ id, label: 'try/except', type: 'error' });
      ctx.edges.push({ from: currentFnId, to: id });
      for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
      return;
    }
  }

  for (const child of node.namedChildren) collectCallsAndFlow(child, ctx, currentFnId);
}

export function extractPythonFlowFromAst(rootNode: SyntaxNode, truncateOnly: boolean): Omit<ParsedFlow, 'language' | 'truncated'> {
  const ctx: WalkCtx = { nodes: [], edges: [], fnIds: new Map(), counter: 0 };
  collectFunctions(rootNode, ctx);
  if (!truncateOnly) collectCallsAndFlow(rootNode, ctx, null);
  return { nodes: ctx.nodes, edges: ctx.edges };
}

export async function parsePython(content: string): Promise<ParsedFlow> {
  const parser = await loadParser();
  await loadLanguage(SupportedLanguages.Python);
  const tree = parser.parse(content);

  const totalNodes = countNodes(tree.rootNode as unknown as SyntaxNode);
  const truncated = totalNodes > MAX_AST_NODES;
  const { nodes, edges } = extractPythonFlowFromAst(tree.rootNode as unknown as SyntaxNode, truncated);
  return { nodes, edges, language: 'python', truncated };
}
```

```typescript
        <div className="border-b border-border-subtle p-3">
          <div className="mb-2 flex items-center gap-1.5">
            <GitFork className="h-3.5 w-3.5 text-accent" />
            <span className="text-xs font-semibold uppercase tracking-wider text-text-muted">
              Code Flow
            </span>
          </div>
          <input
            type="text"
            placeholder="Buscar archivo…"
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="w-full rounded border border-border-subtle bg-surface px-2 py-1.5 text-xs text-text-secondary placeholder-text-muted outline-none focus:border-accent/50"
          />
        </div>

        <div className="flex-1 overflow-y-auto py-1">
          {groupedFiles.size === 0 && (
            <p className="px-3 py-4 text-center text-xs text-text-muted">
              {searchQuery ? 'Sin resultados' : 'No hay archivos compatibles'}
            </p>
          )}
          {Array.from(groupedFiles.entries()).map(([folder, files]) => (
            <div key={folder}>
              <p className="px-3 pb-1 pt-2 text-[9px] uppercase tracking-wider text-text-muted truncate" title={folder}>
                {folder}
              </p>
              {files.map((file) => {
                const ext = getExt(file.filePath);
                const isSelected = selectedFilePath === file.filePath;
                return (
                  <button
                    key={file.filePath}
                    onClick={() => setSelectedFilePath(file.filePath)}
                    className={`flex w-full items-center gap-2 px-3 py-1.5 text-left text-xs transition-colors ${
                      isSelected
                        ? 'border-l-2 border-accent bg-elevated text-accent'
                        : 'border-l-2 border-transparent text-text-secondary hover:bg-hover hover:text-text-primary'
                    }`}
                    title={file.filePath}
                  >
                    <span
                      className="flex-shrink-0 font-mono text-[9px] font-semibold"
                      style={{ color: extColor(ext) }}
                    >
                      {extLabel(ext)}
                    </span>
                    <span className="truncate">{file.name}</span>
                  </button>
                );
              })}
            </div>
          ))}
        </div>

        <div className="border-t border-border-subtle px-3 py-2 text-[10px] text-text-muted">
          {totalCompatible} archivos compatibles
        </div>
      </div>

      {/* Panel derecho — diagrama */}
      <div className="relative flex flex-1 flex-col bg-void">
        {/* Empty state */}
        {!selectedFilePath && !isLoading && (
          <div className="flex h-full flex-col items-center justify-center gap-3 text-center">
            <GitFork className="h-10 w-10 text-text-muted/30" />
            <p className="text-sm text-text-muted">
              Selecciona un archivo para ver su flujo
            </p>
          </div>
        )}

        {/* Loading */}
        {isLoading && (
          <div className="flex h-full items-center justify-center">
            <div className="flex items-center gap-2 text-sm text-text-muted">
              <div className="h-4 w-4 animate-spin rounded-full border-2 border-accent/30 border-t-accent" />
              Analizando…
            </div>
          </div>
        )}

        {/* Truncated banner */}
        {isTruncated && !isLoading && (
          <div className="absolute left-4 top-4 z-10 rounded-lg border border-amber-500/30 bg-amber-500/10 px-3 py-2 text-xs text-amber-300">
            Archivo grande — mostrando solo funciones de nivel superior
          </div>
        )}

        {/* Error */}
        {error && !isLoading && (
          <div className="flex h-full items-center justify-center">
            <div className="max-w-sm rounded-lg border border-rose-500/30 bg-rose-500/10 p-4 text-center text-sm text-rose-300">
              {error}
            </div>
          </div>
        )}

        {/* Export button */}
        {svgHtml && !isLoading && (
          <button
            onClick={handleExport}
            className="absolute right-4 top-4 z-10 flex items-center gap-1.5 rounded-lg border border-border-subtle bg-elevated px-3 py-1.5 text-xs text-text-secondary transition-colors hover:bg-hover hover:text-text-primary"
            title="Exportar como SVG"
          >
            <Download className="h-3.5 w-3.5" />
            Exportar SVG
          </button>
        )}

        {/* Diagram with pan+zoom */}
        {svgHtml && !isLoading && (
          <TransformWrapper ref={transformRef} minScale={0.1} maxScale={5} centerOnInit>
            <TransformComponent
              wrapperStyle={{ width: '100%', height: '100%' }}
              contentStyle={{ width: '100%', height: '100%', display: 'flex', alignItems: 'center', justifyContent: 'center' }}
            >
              <div
                ref={svgRef}
                className="p-8"
                dangerouslySetInnerHTML={{ __html: svgHtml }}
              />
            </TransformComponent>
          </TransformWrapper>
        )}
      </div>
    </div>
  );
});

CodeFlowView.displayName = 'CodeFlowView';
```
