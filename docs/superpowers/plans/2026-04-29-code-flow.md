**Goal:** Añadir una quinta vista "Code Flow" que muestra el flujo de ejecución interno de un archivo seleccionado como flowchart Mermaid interactivo, completamente en el browser.

El botón "Code Flow" se añade al toolbar existente en `GraphCanvas.tsx`.

```typescript
// src/lib/codeflow/graphToMermaid.ts

import type { FlowGraph } from './types';

const MERMAID_STYLES = [
  'classDef fn fill:#1e293b,stroke:#22d3ee,stroke-width:2px,color:#f1f5f9',
  'classDef decision fill:#1e293b,stroke:#f59e0b,stroke-width:2px,color:#f1f5f9',
  'classDef loop fill:#1e293b,stroke:#8b5cf6,stroke-width:2px,color:#f1f5f9',
  'classDef error fill:#1e293b,stroke:#f43f5e,stroke-width:2px,color:#f1f5f9',
];

function sanitizeLabel(text: string): string {
  return text.replace(/"/g, "'").replace(/[<>{}|]/g, '').slice(0, 40);
}

export function graphToMermaid(graph: FlowGraph): string {
  const lines = ['flowchart TD', ...MERMAID_STYLES.map((s) => `  ${s}`)];

  for (const node of graph.nodes) {
    const safe = sanitizeLabel(node.label);
    if (node.kind === 'decision') {
      lines.push(`  ${node.id}{"${safe}"}:::decision`);
    } else {
      lines.push(`  ${node.id}["${safe}"]:::${node.kind}`);
    }
  }

  for (const edge of graph.edges) {
    if (edge.label) {
      lines.push(`  ${edge.from} -->|${edge.label}| ${edge.to}`);
    } else {
      lines.push(`  ${edge.from} --> ${edge.to}`);
    }
  }

  return lines.join('\n');
}
```

```typescript
// src/lib/codeflow/parsers/js-ts.ts

import type { SyntaxNode } from 'web-tree-sitter';
import type { FlowGraph, FlowNode, FlowEdge } from '../types';
import { loadParser, loadLanguage } from '../../../core/tree-sitter/parser-loader';
import { SupportedLanguages } from '../../../config/supported-languages';

function sanitizeId(name: string): string {
  return 'n_' + name.replace(/[^a-zA-Z0-9_]/g, '_').slice(0, 30);
}

interface FnInfo {
  id: string;
  name: string;
  bodyNode: SyntaxNode | null;
}

function countAstNodes(node: SyntaxNode): number {
  let n = 1;
  for (const child of node.namedChildren) n += countAstNodes(child);
  return n;
}

function collectFunctions(root: SyntaxNode): FnInfo[] {
  const fns: FnInfo[] = [];

  function walk(node: SyntaxNode): void {
    switch (node.type) {
      case 'function_declaration':
      case 'generator_function_declaration': {
        const name = node.childForFieldName('name')?.text;
        if (name) fns.push({ id: sanitizeId(name), name, bodyNode: node.childForFieldName('body') });
        break;
      }
      case 'variable_declarator': {
        const name = node.childForFieldName('name')?.text;
        const val = node.childForFieldName('value');
        if (name && val && (val.type === 'arrow_function' || val.type === 'function_expression')) {
          const body = val.childForFieldName('body') ?? val;
          fns.push({ id: sanitizeId(name), name, bodyNode: body });
        }
        break;
      }
      case 'method_definition': {
        const name = node.childForFieldName('name')?.text;
        if (name && name !== 'constructor') {
          const val = node.childForFieldName('value');
          fns.push({ id: sanitizeId(name), name, bodyNode: val?.childForFieldName('body') ?? null });
        }
        break;
      }
    }
    for (const child of node.namedChildren) walk(child);
  }

  walk(root);
  return fns;
}

function collectEdgesAndControlFlow(
  fn: FnInfo,
  knownIds: Set<string>,
  nodes: FlowNode[],
  edges: FlowEdge[],
  counter: { n: number },
): void {
  if (!fn.bodyNode) return;

  function walk(node: SyntaxNode, skipNested: boolean): void {
    // Don't descend into nested function bodies
    if (
      skipNested &&
      (node.type === 'function_declaration' ||
        node.type === 'arrow_function' ||
        node.type === 'function_expression' ||
        node.type === 'method_definition')
    ) return;

    switch (node.type) {
      case 'call_expression': {
        const fnChild = node.childForFieldName('function');
        if (fnChild) {
          const callee =
            fnChild.type === 'identifier'
              ? fnChild.text
              : fnChild.type === 'member_expression'
              ? fnChild.childForFieldName('property')?.text ?? ''
              : '';
          const calleeId = sanitizeId(callee);
          if (callee && knownIds.has(calleeId) && calleeId !== fn.id) {
            // Avoid duplicate edges
            const exists = edges.some((e) => e.from === fn.id && e.to === calleeId);
            if (!exists) edges.push({ from: fn.id, to: calleeId });
          }
        }
        break;
      }
      case 'if_statement': {
        const cond = node.childForFieldName('condition')?.text.slice(0, 25) ?? 'condition?';
        const id = `cf_${counter.n++}`;
        nodes.push({ id, label: cond, kind: 'decision' });
        edges.push({ from: fn.id, to: id });
        const cons = node.childForFieldName('consequence');
        const alt = node.childForFieldName('alternative');
        if (cons) edges.push({ from: id, to: fn.id, label: 'true' });
        if (alt) edges.push({ from: id, to: fn.id, label: 'false' });
        return; // Don't walk further into if body (to avoid duplicate edges)
      }
      case 'for_statement':
      case 'for_in_statement':
      case 'for_of_statement': {
        const id = `cf_${counter.n++}`;
        nodes.push({ id, label: '🔁 for loop', kind: 'loop' });
        edges.push({ from: fn.id, to: id });
        return;
      }
      case 'while_statement': {
        const id = `cf_${counter.n++}`;
        nodes.push({ id, label: '🔁 while', kind: 'loop' });
        edges.push({ from: fn.id, to: id });
        return;
      }
      case 'try_statement': {
        const id = `cf_${counter.n++}`;
        nodes.push({ id, label: '⚠️ try/catch', kind: 'error' });
        edges.push({ from: fn.id, to: id });
        return;
      }
    }

    for (const child of node.namedChildren) walk(child, true);
  }

  walk(fn.bodyNode, true);
}

export async function parseJsTs(content: string, filePath: string): Promise<FlowGraph> {
  const parser = await loadParser();
  const lang = filePath.endsWith('.tsx') ? SupportedLanguages.TypeScript : 
               filePath.endsWith('.ts') || filePath.endsWith('.tsx') ? SupportedLanguages.TypeScript :
               SupportedLanguages.JavaScript;
  await loadLanguage(lang, filePath);

  const tree = parser.parse(content);
  const root = tree.rootNode;

  const nodeCount = countAstNodes(root);
  const truncated = nodeCount > 500;

  const rawFns = collectFunctions(root);
  // Si truncado, solo funciones top-level (profundidad ≤ 2 del root)
  const fns = truncated ? rawFns.slice(0, 15) : rawFns;

  const flowNodes: FlowNode[] = fns.map((fn) => ({
    id: fn.id,
    label: fn.name,
    kind: 'fn',
  }));

  const knownIds = new Set(fns.map((fn) => fn.id));
  const flowEdges: FlowEdge[] = [];
  const counter = { n: 0 };

  if (!truncated) {
    for (const fn of fns) {
      collectEdgesAndControlFlow(fn, knownIds, flowNodes, flowEdges, counter);
    }
  } else {
    // En modo truncado, solo aristas entre funciones (sin control flow)
    for (const fn of fns) {
      if (!fn.bodyNode) continue;
      const findCalls = (node: SyntaxNode): void => {
        if (node.type === 'call_expression') {
          const fnChild = node.childForFieldName('function');
          if (fnChild) {
            const callee = fnChild.type === 'identifier' ? fnChild.text :
                          fnChild.type === 'member_expression' ? fnChild.childForFieldName('property')?.text ?? '' : '';
            const calleeId = sanitizeId(callee);
            if (callee && knownIds.has(calleeId) && calleeId !== fn.id) {
              const exists = flowEdges.some(e => e.from === fn.id && e.to === calleeId);
              if (!exists) flowEdges.push({ from: fn.id, to: calleeId });
            }
          }
        }
        for (const child of node.namedChildren) findCalls(child);
      };
      findCalls(fn.bodyNode);
    }
  }

  // Eliminar nodos huérfanos (fn nodes sin ninguna arista) si hay más de 8 funciones
  let finalNodes = flowNodes;
  let finalEdges = flowEdges;
  if (fns.length > 8) {
    const connected = new Set<string>();
    for (const e of flowEdges) { connected.add(e.from); connected.add(e.to); }
    finalNodes = flowNodes.filter(n => connected.has(n.id));
    finalEdges = flowEdges.filter(e => finalNodes.some(n => n.id === e.from));
  }

  return { nodes: finalNodes, edges: finalEdges, truncated };
}
```

```typescript
import type { SyntaxNode } from 'web-tree-sitter';
import type { FlowGraph, FlowNode, FlowEdge } from '../types';
import { loadParser, loadLanguage } from '../../../core/tree-sitter/parser-loader';
import { SupportedLanguages } from '../../../config/supported-languages';

function sanitizeId(name: string): string {
  return 'n_' + name.replace(/[^a-zA-Z0-9_]/g, '_').slice(0, 30);
}

function countAstNodes(node: SyntaxNode): number {
  let n = 1;
  for (const child of node.namedChildren) n += countAstNodes(child);
  return n;
}

interface FnInfo {
  id: string;
  name: string;
  bodyNode: SyntaxNode | null;
}

function collectFunctions(root: SyntaxNode): FnInfo[] {
  const fns: FnInfo[] = [];

  function walk(node: SyntaxNode): void {
    if (node.type === 'function_definition' || node.type === 'async_function_definition') {
      const name = node.childForFieldName('name')?.text;
      if (name) {
        fns.push({ id: sanitizeId(name), name, bodyNode: node.childForFieldName('body') });
      }
    }
    // decorated_definition wraps functions with @decorator
    if (node.type === 'decorated_definition') {
      const def = node.childForFieldName('definition');
      if (def && (def.type === 'function_definition' || def.type === 'async_function_definition')) {
        const name = def.childForFieldName('name')?.text;
        if (name) {
          fns.push({ id: sanitizeId(name), name, bodyNode: def.childForFieldName('body') });
        }
      }
    }
    for (const child of node.namedChildren) walk(child);

  return fns;

        flowEdges.push({ from: fn.id, to: id });
        const hasElse = node.namedChildren.some((c) => c.type === 'else_clause');
        flowEdges.push({ from: id, to: fn.id, label: 'true' });
        if (hasElse) flowEdges.push({ from: id, to: fn.id, label: 'false' });
        return;
      }
      case 'for_statement': {
        const id = `cf_${counter.n++}`;
        flowNodes.push({ id, label: '🔁 for loop', kind: 'loop' });

}
```

```typescript
// src/lib/codeflow/parsers/markup.ts

import type { FlowGraph } from '../types';

// ── HTML ────────────────────────────────────────────────────────────────────

const SKIP_TAGS = new Set(['head', 'meta', 'link', 'script', 'style', 'title', 'base', 'noscript']);

export function parseHtml(content: string): FlowGraph {
  const parser = new DOMParser();
  const doc = parser.parseFromString(content, 'text/html');
  const nodes: FlowGraph['nodes'] = [];
  const edges: FlowGraph['edges'] = [];
  let counter = 0;

  function walk(el: Element, parentId: string | null, depth: number): void {
    if (depth > 5) return;
    const tag = el.tagName.toLowerCase();
    if (SKIP_TAGS.has(tag)) return;

    const id = `el_${counter++}`;
    const idAttr = el.getAttribute('id');
    const classAttr = el.className ? `.${el.className.split(' ')[0]}` : '';
    const label = idAttr ? `<${tag}#${idAttr}>` : classAttr ? `<${tag}${classAttr}>` : `<${tag}>`;

    nodes.push({ id, label, kind: 'fn' });
    if (parentId) edges.push({ from: parentId, to: id });

    for (const child of Array.from(el.children)) {
      walk(child, id, depth + 1);
    }
  }

  walk(doc.body ?? doc.documentElement, null, 0);
  return { nodes, edges };
}

// ── CSS ─────────────────────────────────────────────────────────────────────

export function parseCss(content: string): FlowGraph {
  const nodes: FlowGraph['nodes'] = [];
  // Match selectors: everything before { on a non-empty line
  const rulePattern = /^\s*([^{/@][^{]*?)\s*\{/gm;
  let match: RegExpExecArray | null;
  let counter = 0;
  const seen = new Set<string>();

  while ((match = rulePattern.exec(content)) !== null) {
    const raw = match[1].trim();
    // Split comma-separated selectors
    for (const sel of raw.split(',')) {
      const label = sel.trim().slice(0, 40);
      if (!label || seen.has(label)) continue;
      seen.add(label);
      nodes.push({ id: `sel_${counter++}`, label, kind: 'fn' });
      if (nodes.length >= 30) return { nodes, edges: [] };
    }
  }

  return { nodes, edges: [] };
}

// ── JSON ────────────────────────────────────────────────────────────────────

// ── YAML ────────────────────────────────────────────────────────────────────

export function parseYaml(content: string): FlowGraph {
  const nodes: FlowGraph['nodes'] = [];
  const edges: FlowGraph['edges'] = [];
  let counter = 0;
  const stack: Array<{ id: string; indent: number }> = [];
  const keyPattern = /^(\s*)([a-zA-Z_][a-zA-Z0-9_-]*)\s*:/gm;
  let match: RegExpExecArray | null;

  while ((match = keyPattern.exec(content)) !== null) {
    const indent = match[1].length;
    const key = match[2];
    const id = `key_${counter++}`;
    nodes.push({ id, label: key.slice(0, 30), kind: 'fn' });
    while (stack.length > 0 && stack[stack.length - 1].indent >= indent) stack.pop();
    if (stack.length > 0) edges.push({ from: stack[stack.length - 1].id, to: id });
    stack.push({ id, indent });
    if (nodes.length >= 40) break;
  }

  return { nodes, edges };
}

// ── JSON ────────────────────────────────────────────────────────────────────

export function parseJson(content: string): FlowGraph {
  let parsed: unknown;
  try {
    parsed = JSON.parse(content);
  } catch {
    return { nodes: [{ id: 'err', label: 'JSON inválido', kind: 'error' }], edges: [] };
  }

  const nodes: FlowGraph['nodes'] = [];
  const edges: FlowGraph['edges'] = [];
  let counter = 0;

  function walk(value: unknown, parentId: string | null, key: string, depth: number): void {
    if (depth > 3) return;
    const id = `key_${counter++}`;
    const label = depth === 0 ? '{ root }' : key;
    const kind: FlowGraph['nodes'][0]['kind'] =
      typeof value === 'object' && value !== null && !Array.isArray(value) ? 'fn' :
      Array.isArray(value) ? 'loop' : 'fn';
    nodes.push({ id, label: label.slice(0, 30), kind });
    if (parentId) edges.push({ from: parentId, to: id });

    if (typeof value === 'object' && value !== null) {
      const entries = Array.isArray(value)
        ? value.slice(0, 10).map((v, i) => [`[${i}]`, v] as [string, unknown])
        : Object.entries(value as Record<string, unknown>).slice(0, 20);
      for (const [k, v] of entries) walk(v, id, k, depth + 1);
    }
  }

  walk(parsed, null, 'root', 0);
  return { nodes, edges };
}
```

```typescript
// src/lib/codeflow/astToMermaid.ts

import type { AstToMermaidResult } from './types';
import { graphToMermaid } from './graphToMermaid';

type SupportedExt = 'js' | 'jsx' | 'mjs' | 'cjs' | 'ts' | 'tsx' | 'py' | 'html' | 'htm' | 'css' | 'scss' | 'less' | 'json' | 'yaml' | 'yml';

function getExt(filePath: string): string {
  return filePath.split('.').pop()?.toLowerCase() ?? '';
}

function isSupportedExt(ext: string): ext is SupportedExt {
  return ['js', 'jsx', 'mjs', 'cjs', 'ts', 'tsx', 'py', 'html', 'htm', 'css', 'scss', 'less', 'json', 'yaml', 'yml'].includes(ext);
}

export async function astToMermaid(
  filePath: string,
  content: string,
): Promise<AstToMermaidResult | null> {
  const ext = getExt(filePath);
  if (!isSupportedExt(ext)) return null;

  // JS/TS family
  if (['js', 'jsx', 'mjs', 'cjs', 'ts', 'tsx'].includes(ext)) {
    const { parseJsTs } = await import('./parsers/js-ts');
    const graph = await parseJsTs(content, filePath);
    if (graph.nodes.length === 0) {
      return { mermaid: 'flowchart TD\n  A["Sin funciones detectadas"]', truncated: false };
    }
    return { mermaid: graphToMermaid(graph), truncated: graph.truncated ?? false };
  }

  // Python
  if (ext === 'py') {
    const { parsePython } = await import('./parsers/python');
    const graph = await parsePython(content);
    if (graph.nodes.length === 0) {
      return { mermaid: 'flowchart TD\n  A["Sin funciones detectadas"]', truncated: false };
    }
    return { mermaid: graphToMermaid(graph), truncated: graph.truncated ?? false };
  }

  // HTML
  if (ext === 'html' || ext === 'htm') {
    const { parseHtml } = await import('./parsers/markup');
    const graph = parseHtml(content);
    return { mermaid: graphToMermaid(graph), truncated: false };
  }

  // CSS family
  if (ext === 'css' || ext === 'scss' || ext === 'less') {
    const { parseCss } = await import('./parsers/markup');
    const graph = parseCss(content);
    if (graph.nodes.length === 0) {
      return { mermaid: 'flowchart TD\n  A["Sin selectores detectados"]', truncated: false };
    }
    return { mermaid: graphToMermaid(graph), truncated: false };
  }

  // JSON
  if (ext === 'json') {
    const { parseJson } = await import('./parsers/markup');
    const graph = parseJson(content);
    return { mermaid: graphToMermaid(graph), truncated: false };
  }

  // YAML
  if (ext === 'yaml' || ext === 'yml') {
    const { parseYaml } = await import('./parsers/markup');
    const graph = parseYaml(content);
    if (graph.nodes.length === 0) {
      return { mermaid: 'flowchart TD\n  A["Sin claves detectadas"]', truncated: false };
    }
    return { mermaid: graphToMermaid(graph), truncated: false };
  }

  return null;
}
```

```tsx
export const CodeFlowView = forwardRef<CodeFlowViewHandle>((_, ref) => {
  const { selectedNode } = useAppState();
  const transformRef = useRef<ReactZoomPanPinchRef>(null);

  const [viewState, setViewState] = useState<ViewState>('idle');
  const [svg, setSvg] = useState('');
  const [errorMsg, setErrorMsg] = useState('');
  const [truncated, setTruncated] = useState(false);
  const [unsupportedExt, setUnsupportedExt] = useState('');

  useImperativeHandle(ref, () => ({
    zoomIn: () => transformRef.current?.zoomIn(),
    zoomOut: () => transformRef.current?.zoomOut(),
    resetZoom: () => transformRef.current?.resetTransform(),
  }));

      </div>
    );
  }

  // ── Parsing ───────────────────────────────────────────────────────────────
  if (viewState === 'parsing') {
    return (
      <div className="flex h-full w-full flex-col items-center justify-center gap-3 bg-void">

        <p className="text-sm text-text-muted">Analizando flujo…</p>
      </div>

  }

  // ── Unsupported ───────────────────────────────────────────────────────────

    return (
      <div className="flex h-full w-full flex-col items-center justify-center gap-3 bg-void">
        <Share2 className="h-8 w-8 text-text-muted opacity-40" />
        <p className="text-sm text-text-muted">

        </p>
      </div>
    );
  }

  // ── Error ─────────────────────────────────────────────────────────────────
  if (viewState === 'error') {
    return (
      <div className="flex h-full w-full flex-col items-center justify-center gap-3 bg-void px-8">
        <AlertTriangle className="h-6 w-6 text-rose-400" />
        <p className="text-sm text-rose-300">Error al generar el diagrama</p>
        <details className="mt-1">
          <summary className="cursor-pointer text-xs text-text-muted hover:text-text-secondary">Ver detalle</summary>

        </details>
      </div>
    );

  return (
```
